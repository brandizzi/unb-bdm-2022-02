 DROP DATABASE IF EXISTS "VIAGENSBICICLETA";

CREATE DATABASE "VIAGENSBICICLETA"
    WITH
    OWNER = postgres
    ENCODING = 'UTF8'
    LC_COLLATE = 'UTF8'
    LC_CTYPE = 'UTF8'
    TABLESPACE = pg_default
    CONNECTION LIMIT = -1
    IS_TEMPLATE = False;

DROP SCHEMA IF EXISTS "VIAGENSBICICLETAS" ;

CREATE SCHEMA IF NOT EXISTS "VIAGENSBICICLETAS"
    AUTHORIZATION postgres;
